Endpoints Principales

GET/api/v1/leads
// Obtener leads con filtros
curl -X GET "https://api.sisteco.io/v1/leads?score_min=80&limit=50" \
  -H "Authorization: Bearer sk_live_..."
POST/api/v1/campaigns
// Crear una nueva campaña
curl -X POST "https://api.sisteco.io/v1/campaigns" \
  -H "Authorization: Bearer sk_live_..." \
  -H "Content-Type: application/json" \
  -d '{"name": "Q1 Outreach", "channels": ["email", "linkedin"]}'
JavaScriptSDK Example
import { Sisteco } from '@sisteco/sdk';

const client = new Sisteco('sk_live_...');

const analytics = await client.analytics.get({
  period: 'last_30_days',
  metrics: ['leads', 'open_rate', 'meetings']
});
